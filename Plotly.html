Load the data

data("rest_inspec")

Make a histogram about the number of grade A, B, and C restaurants

rest_inspec %>%
  select(boro, grade) %>% 
  filter(grade == c("A", "B", "C")) %>% 
  group_by(boro, grade) %>% 
  filter(boro !="Missing") %>% 
  summarize(
    count = n()
  ) %>% 
  plot_ly(
    x = ~boro, 
    y = ~count,
    color = ~grade,
    type = "bar",
    colors = "viridis") 
## `summarise()` has grouped output by 'boro'. You can override using the `.groups` argument.

Make boxplot about score of grade A restaurant in different locations

rest_inspec %>% 
  select(boro, score, grade) %>% 
  group_by(boro) %>% 
  filter(score != "NA",
         grade == "A",
         boro != "Missing") %>% 
  plot_ly(
    y = ~score,
    color =~boro,
    type = "box",
    colors = "viridis"
  )

Make a scatter plot of status of critical flage for restaurants with different grades

rest_inspec %>% 
  select(boro, critical_flag, grade) %>% 
  filter(boro == "MANHATTAN") %>% 
  group_by(grade, critical_flag) %>% 
  summarize(
    count = n()
  ) %>% 
  filter(grade !="Not Yet Graded") %>% 
  mutate(
    critical_flag = forcats::fct_reorder(critical_flag, count))%>% 
  plot_ly(
    x =~critical_flag,
    y =~count,
    color = ~grade,
    type = "scatter",
    mode = "markers"
  )
## `summarise()` has grouped output by 'grade'. You can override using the `.groups` argument.
